### 강의 정리 - 204. 자바 스크립트 적용하기 2

<br />

```html
<body>
  <svg class="face" xmlns="http://www.w3.org/2000/svg" viewBox="0 0 571 625.5">
    // style 넣기
    <style>
      .face {
          width: 200px;
          // 가운데 정렬
          position: absolute;
          top: 0;
          bottom: 0;
          left: 0;
          right: 0;
          margin: auto;
      }

          .face_hair {
              fill: brown; // 색칠하기
          }
          .face_hair:nth-of-type(1) {
              fill: blue;
          }
          .face_eye {
              transform-origin: 50% 50% // center center
          }
          .ani-on .face_eye { //click하면 애니메이션 실행
              animation: eye-ani 0.5s infinite alternate;
          }

          @ keyframes eye-ani {
              0% { transform: scale(0.9); }
              100% { transform: scale(1.1); }
          }
    </style>

    // script 넣기
    <script>
      const faceElem = document.querySelector('.face');
      faceElem.addEventListenr('click', function () {
      this.classList.toggle('ani-on'); });
    </script>

    <circle
      cx="286.5"
      cy="354.5"
      r="251"
      stroke-width="40"
      fill="none"
      stroke="#000"
      stroke-miterlimit="10"
    />
    <circle class="face_eye" cx="175" cy="380.5" r="42.5" />
    <circle class="face_eye" cx="397" cy="380.5" r="42.5" />
    <path
      d="M422 476.16a181.3 181.3 0 0 1-135.48 60.59H286a181.3 181.3 0 0 1-135.48-60.59"
      stroke-width="30"
      fill="none"
      stroke="#000"
      stroke-miterlimit="10"
    />
    <ellipse class="face_hair" cx="168.5" cy="210.5" rx="168.5" ry="121.5" />
    <ellipse class="face_hair" cx="428.5" cy="198.5" rx="142.5" ry="123.5" />
    <ellipse class="face_hair" cx="245.5" cy="103" rx="151.5" ry="103" />
  </svg>
</body>
```
